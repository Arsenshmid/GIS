import pandas as pd
import numpy as np

# Создаем пустой DataFrame
df = pd.DataFrame(columns=['latitude', 'longitude', 'weight', 'irrigated', 'crop'])

# Задаем границы области внутри Якутска (примерно 2 км)
min_lat, max_lat = 62, 62.01
min_lon, max_lon = 129, 129.01

# Задаем шаг для генерации точек
step = 0.00018  # примерно 20 метров

# Заполняем DataFrame данными
i = 0
for lat in np.arange(min_lat, max_lat, step):
    for lon in np.arange(min_lon, max_lon, step):
        weight = np.random.randint(56, 66)  # вес урожая в кг (ближе к предыдущему результату)
        irrigated = np.random.choice([True, False])  # орошалась ли область
        crop = np.random.choice(['wheat', 'corn', 'soy'])  # какая культура была посажена

        df.loc[i] = [lat, lon, weight, irrigated, crop]
        i += 1

# Сохраняем DataFrame в файл CSV
df.to_csv('synthetic_data.csv', index=False)

import folium
from django.shortcuts import render
import pandas as pd
from .models import HarvestData
import numpy as np
import numpy as np

def load_data():
    # Чтение данных из CSV-файла
    df = pd.read_csv('synthetic_data.csv')

    # Генерация равномерно распределенных точек в пределах заданной области
    # Якутск имеет широту примерно от 62 до 63 градусов и долготу от 129 до 130 градусов
    # Поэтому для создания маленькой области внутри Якутии можно использовать близкие значения широты и долготы
    min_lat, max_lat = 62.01, 62.02  # Примерная область широты
    min_lon, max_lon = 129.01, 129.02  # Примерная область долготы
    step = 0.00018  # Примерно 20 метров между точками

    latitudes = np.arange(min_lat, max_lat, step)
    longitudes = np.arange(min_lon, max_lon, step)

    # Загрузка данных в модель Django
    for lat, lon in zip(latitudes, longitudes):
        row = df.sample()  # Выбираем случайную строку из DataFrame
        data = HarvestData(latitude=lat, longitude=lon, weight=row['weight'].values[0], irrigated=row['irrigated'].values[0], crop=row['crop'].values[0])
        data.save()

def home(request):
    # Загрузка данных
    load_data()

    # Создание карты
    m = folium.Map(location=[62.015, 129.015], zoom_start=15)  # Центральная точка области

    # Добавление данных из модели Django на карту
    for data in HarvestData.objects.all():
        # Выбор цвета в зависимости от веса урожая
        if data.weight < 60:
            color = 'red'
        elif data.weight < 70:
            color = 'yellow'
        else:
            color = 'green'

        # Добавление маркера на карту
        folium.CircleMarker([data.latitude, data.longitude], radius=10, popup=f'Weight: {data.weight}', color=color, fill=True, fill_color=color).add_to(m)

    # Генерация HTML-строки с картой
    m = m._repr_html_()

    return render(request, 'home.html', {'map': m})
